#include <iostream>

int main()
{
    int number = 10;

    // Guard clause: если число 5, выводим и выходим
    if (number == 5)
    {
        std::cout << "The number is 5" << std::endl;
        return 0; // ранний выход
    }

    // Иначе число не 5
    std::cout << "The number is not 5" << std::endl;

    // Проверка на нечётность
    if (number % 2 == 1)
    {
        std::cout << "The number is not divisible by 2" << std::endl;
        return 0; // ранний выход
    }

    // Число чётное
    std::cout << "The number is divisible by 2" << std::endl;

    // Дополнительная проверка
    if (number == 6)
    {
        std::cout << "The number is 6" << std::endl;
        return 0;
    }

    // Можно добавить здесь любой код для остальных чисел
}
